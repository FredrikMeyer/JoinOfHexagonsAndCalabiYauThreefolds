\begin{frame}
\frametitle{Mirror symmetry}

\begin{itemize}
	\item Calabi--Yau threefolds seem to ``always'' have ``mirror partners''.
	\item Mirror partner $X^\circ$ to $X$ have ``mirrored Hodge diamond''.
	\item Hence $\chi(X^\circ)= - \chi(X)$.
\end{itemize}
