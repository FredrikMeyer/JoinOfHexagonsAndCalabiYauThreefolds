\frametitle{The cone over $\dP6$}

\begin{itemize}
	\item Let $\dP6 \subset \P^6$ be an anticanonically embedded del Pezzo surface of degree $6$. Let $C(\dP6)$ be its affine cone in $\mathbb A^7$.

	\item The equations are 
	$$
\begin{vmatrix}
y & x_1 & x_2 \\
x_4 & y & x_3 \\
x_ 5 & x_6 & y
\end{vmatrix}
\leq 1
	$$
\end{itemize}
